import { Card, CardGrid } from '@astrojs/starlight/components';
import { getCollection } from 'astro:content';

// 获取所有文章并按日期排序
const allPosts = await getCollection('docs');
const recentPosts = allPosts
  .filter(post => !post.data.template)
  .sort((a, b) => new Date(b.data.pubDate).getTime() - new Date(a.data.pubDate).getTime())
  .slice(0, 6);

// 获取所有分类
const categories = [];
for (const post of allPosts) {
  if (post.data.tags) {
    for (const tag of post.data.tags) {
      if (!categories.includes(tag)) {
        categories.push(tag);
      }
    }
  }
}

## 最新文章

<CardGrid>
  {recentPosts.map(post => (
    <Card title={post.data.title} href={post.slug} icon="document">
      {post.data.description}
      <div style="margin-top: 0.5rem; font-size: 0.875rem; color: var(--sl-color-text-2);">
        {new Date(post.data.pubDate).toLocaleDateString('zh-CN')}
        {post.data.tags && (
          <div style="margin-top: 0.25rem;">
            {post.data.tags.map(tag => (
              <span style="display: inline-block; margin-right: 0.5rem; padding: 0.125rem 0.5rem; background: var(--sl-color-accent-low); border-radius: 0.25rem; font-size: 0.75rem;">
                {tag}
              </span>
            ))}
          </div>
        )}
      </div>
    </Card>
  ))}
</CardGrid>

## 文章分类

<CardGrid>
  {categories.map(category => (

      查看所有 {category} 相关的文章
    </Card>
  ))}
</CardGrid>
